Sub CADASTRO_ENTRADA_SAIDA()
  
    ''Não permitir cadastrar faltando dados
     
     If Range("f7") = Empty Or Range("h7") = Empty Or Range("b10") = Empty Or Range("h10") = Empty Or Range("b13") = Empty Or Range("d13") = Empty Or Range("h13") = Empty Or Range("c7") = Empty Then
     MsgBox ("Preencha todos os campos")
     Exit Sub
     End If
     
     'Desabilitando a atualização tela pique ao gravar'
    
    Application.ScreenUpdating = False
    
    Sheets("ENTRADA | SAIDA").Select
    Range("B8").EntireRow.Insert
    
    ''Copia codigo (ID)
            
    Sheets("CADASTRO").Select
    Range("C7").Copy
    Sheets("ENTRADA | SAIDA").Select
    Range("B8").PasteSpecial xlPasteValues
    
    ''Copia operador
            
    Sheets("CADASTRO").Select
    Range("F7").Copy
    Sheets("ENTRADA | SAIDA").Select
    Range("J8").PasteSpecial xlPasteValues
    
    ''Copia produto
    
    Sheets("CADASTRO").Select
    Range("B10").Copy
    Sheets("ENTRADA | SAIDA").Select
    Range("C8").PasteSpecial xlPasteValues
    
    ''Copia unidade medida
    
    Sheets("CADASTRO").Select
    Range("H10").Copy
    Sheets("ENTRADA | SAIDA").Select
    Range("D8").PasteSpecial xlPasteValues
    
    
    ''Copia quantidade
            
    Sheets("CADASTRO").Select
    Range("B13").Copy
    Sheets("ENTRADA | SAIDA").Select
    Range("E8").PasteSpecial xlPasteValues
    
    ''Copia valor da unidade
        
    Sheets("CADASTRO").Select
    Range("D13").Copy
    Sheets("ENTRADA | SAIDA").Select
    Range("F8").PasteSpecial xlPasteValues
    
    ''Copia total
            
    Sheets("CADASTRO").Select
    Range("F13").Copy
    Sheets("ENTRADA | SAIDA").Select
    Range("G8").PasteSpecial xlPasteValues
    
    ''Copia situação
            
    Sheets("CADASTRO").Select
    Range("H13").Copy
    Sheets("ENTRADA | SAIDA").Select
    Range("H8").PasteSpecial xlPasteValues
    
    ''Data. Cadastro
            
    Sheets("CADASTRO").Select
    Range("H7").Copy
    Sheets("ENTRADA | SAIDA").Select
    Range("I8").PasteSpecial xlPasteValuesAndNumberFormats
    
    ''Copia fornecedor
            
    Sheets("CADASTRO").Select
    Range("F16").Copy
    Sheets("ENTRADA | SAIDA").Select
    Range("K8").PasteSpecial xlPasteValues
    
    ''Copia nº nota
            
    Sheets("CADASTRO").Select
    Range("D16").Copy
    Sheets("ENTRADA | SAIDA").Select
    Range("L8").PasteSpecial xlPasteValues
    
   
    ''Apagando depois de cadastrar

    Sheets("CADASTRO").Select
    Range("F7").ClearContents 'Operador
    Range("H7").ClearContents 'Data cadastro
    Range("B10").ClearContents 'Produto
    Range("B13").ClearContents 'qtd
    Range("D13").ClearContents 'valor unid
    Range("H13").ClearContents 'Situação
    Range("H10").ClearContents 'unidade Medida
    Range("D16").ClearContents 'nº nota
    Range("F16").ClearContents 'fornecedor
    
        
    ''Mensagem de cadastro
    
    MsgBox ("Cadastro realizado com sucesso!")
    
              
End Sub
