Sub BAIXA_ESTOQUE_CESTA()

  
    ''Não permitir cadastrar faltando dados
     
     If Range("C9") = Empty Or Range("F9") = Empty Or Range("J9") = Empty Then
     MsgBox ("Preencha situação, data e operador.")
     Exit Sub
     End If
     
     'Desabilitando a atualização tela pique ao gravar'
    
    Application.ScreenUpdating = False
    
     '' POSTO 01 CIDADE DOS MENINOS
    
    Sheets("CESTAS MONTADAS").Select
    Range("B9").EntireRow.Insert
    
    ''Copia posto(POSTO ABASTECIMENTO)
            
    Sheets("SEPARAÇÃO DE CESTAS").Select
    Range("B13").Copy
    Sheets("CESTAS MONTADAS").Select
    Range("AF9").PasteSpecial xlPasteValues
    
    
    
    ''Copia codigo (ID)
            
    Sheets("SEPARAÇÃO DE CESTAS").Select
    Range("C7").Copy
    Sheets("CESTAS MONTADAS").Select
    Range("B9").PasteSpecial xlPasteValues
    
   ''Copia lote (ID LOTE)
            
    Sheets("SEPARAÇÃO DE CESTAS").Select
    Range("H7").Copy
    Sheets("CESTAS MONTADAS").Select
    Range("C9").PasteSpecial xlPasteValues
    
     
    ''Copia situação
            
    Sheets("SEPARAÇÃO DE CESTAS").Select
    Range("C9").Copy
    Sheets("CESTAS MONTADAS").Select
    Range("AE9").PasteSpecial xlPasteValues
    
    ''Data. Cadastro
            
    Sheets("SEPARAÇÃO DE CESTAS").Select
    Range("F9").Copy
    Sheets("CESTAS MONTADAS").Select
    Range("D9").PasteSpecial xlPasteValuesAndNumberFormats
    
   ''Copia operador
    
    Sheets("SEPARAÇÃO DE CESTAS").Select
    Range("J9").Copy
    Sheets("CESTAS MONTADAS").Select
    Range("E9").PasteSpecial xlPasteValues
    
    ''copia posto 01 - CIDADE DOS MENINOS
          
    ''Copia cesta 01
    
    Sheets("SEPARAÇÃO DE CESTAS").Select
    Range("B15").Copy
    Sheets("CESTAS MONTADAS").Select
    Range("G9").PasteSpecial xlPasteValues
    
    
    ''Copia cesta 02
            
    Sheets("SEPARAÇÃO DE CESTAS").Select
    Range("C15").Copy
    Sheets("CESTAS MONTADAS").Select
    Range("H9").PasteSpecial xlPasteValues
    
    ''Copia cesta 03
            
    Sheets("SEPARAÇÃO DE CESTAS").Select
    Range("D15").Copy
    Sheets("CESTAS MONTADAS").Select
    Range("I9").PasteSpecial xlPasteValues
    
     ''Copia cesta 01/02
    
    Sheets("SEPARAÇÃO DE CESTAS").Select
    Range("B17").Copy
    Sheets("CESTAS MONTADAS").Select
    Range("J9").PasteSpecial xlPasteValues
    
    
    ''Copia cesta 02/02
            
    Sheets("SEPARAÇÃO DE CESTAS").Select
    Range("C17").Copy
    Sheets("CESTAS MONTADAS").Select
    Range("K9").PasteSpecial xlPasteValues
    
    ''Copia total
            
    Sheets("SEPARAÇÃO DE CESTAS").Select
    Range("C19").Copy
    Sheets("CESTAS MONTADAS").Select
    Range("M9").PasteSpecial xlPasteValues

End Sub
